251     private String getOutOfProcessJavacVersion(String executable) throws CompilerException {
252         String version = VERSION_PER_EXECUTABLE.get(executable);
253         if (version == null) {
254             Commandline cli = new Commandline();
255             cli.setExecutable(executable);
256             /*
257              * The option "-version" should be supported by javac since 1.6 (https://docs.oracle.com/javase/6/docs/technotes/tools/solaris/javac.html)
258              * up to 21 (https://docs.oracle.com/en/java/javase/21/docs/specs/man/javac.html#standard-options)
259              */
260             cli.addArguments(new String[] {"-version"}); //
261             CommandLineUtils.StringStreamConsumer out = new CommandLineUtils.StringStreamConsumer();
262             try {
263                 int exitCode = CommandLineUtils.executeCommandLine(cli, out, out);
264                 if (exitCode != 0) {
265                     throw new CompilerException("Could not retrieve version from " + executable + ". Exit code "
266                             + exitCode + ", Output: " + out.getOutput());
267                 }
268             } catch (CommandLineException e) {
269                 throw new CompilerException("Error while executing the external compiler " + executable, e);
270             }
271             version = extractMajorAndMinorVersion(out.getOutput());
272             VERSION_PER_EXECUTABLE.put(executable, version);
273         }
274         return version;
275     }
276 
277     static String extractMajorAndMinorVersion(String text) {
278         Matcher matcher = JAVA_MAJOR_AND_MINOR_VERSION_PATTERN.matcher(text);
279         if (!matcher.find()) {
280             throw new IllegalArgumentException("Could not extract version from \"" + text + "\"");
281         }
282         return matcher.group();
283     }

285     @Override
286     public CompilerResult performCompile(CompilerConfiguration config) throws CompilerException {
287         File destinationDir = new File(config.getOutputLocation());
288         if (!destinationDir.exists()) {
289             destinationDir.mkdirs();
290         }
291 
292         String[] sourceFiles = getSourceFiles(config);
293         if ((sourceFiles == null) || (sourceFiles.length == 0)) {
294             return new CompilerResult();
295         }
296 
297         logCompiling(sourceFiles, config);
298 
299         final String javacVersion;
300         final String executable;
301         if (config.isFork()) {
302             executable = getJavacExecutable(config);
303             javacVersion = getOutOfProcessJavacVersion(executable);
304         } else {
305             javacVersion = getInProcessJavacVersion();
306             executable = null;
307         }
308 
309         String[] args = buildCompilerArguments(config, sourceFiles, javacVersion);
310         CompilerResult result;
311 
312         if (config.isFork()) {
313             result = compileOutOfProcess(config, executable, args);
314         } else {
315             if (hasJavaxToolProvider() && !config.isForceJavacCompilerUse()) {
316                 // use fqcn to prevent loading of the class on 1.5 environment !
317                 result = inProcessCompiler().compileInProcess(args, config, sourceFiles);
318             } else {
319                 result = compileInProcess(args, config);
320             }
321         }
322 
323         return result;
324     }

344         final String javacVersion;
345         if (config.isFork()) {
346             String executable = getJavacExecutable(config);
347             javacVersion = getOutOfProcessJavacVersion(executable);
348         } else {
349             javacVersion = getInProcessJavacVersion();
350         }
351         return buildCompilerArguments(config, getSourceFiles(config), javacVersion);

354     public static String[] buildCompilerArguments(
355             CompilerConfiguration config, String[] sourceFiles, String javacVersion) {
356         List<String> args = new ArrayList<>();
357 
358         // ----------------------------------------------------------------------
359         // Set output
360         // ----------------------------------------------------------------------
361 
362         File destinationDir = new File(config.getOutputLocation());
363         args.add("-d");
364         args.add(destinationDir.getAbsolutePath());
365 
366         // ----------------------------------------------------------------------
367         // Set the class and source paths
368         // ----------------------------------------------------------------------
369 
370         List<String> classpathEntries = config.getClasspathEntries();
371         if (classpathEntries != null && !classpathEntries.isEmpty()) {
372             args.add("-classpath");
373             args.add(getPathString(classpathEntries));
374         }
375 
376         List<String> modulepathEntries = config.getModulepathEntries();
377         if (modulepathEntries != null && !modulepathEntries.isEmpty()) {
378             args.add("--module-path");
379             args.add(getPathString(modulepathEntries));
380         }
381 
382         List<String> sourceLocations = config.getSourceLocations();
383         if (sourceLocations != null && !sourceLocations.isEmpty()) {
384             // always pass source path, even if sourceFiles are declared,
385             // needed for jsr269 annotation processing, see MCOMPILER-98
386             args.add("-sourcepath");
387             args.add(getPathString(sourceLocations));
388         }
389         if (!hasJavaxToolProvider() || config.isForceJavacCompilerUse() || config.isFork()) {
390             args.addAll(Arrays.asList(sourceFiles));
391         }
392 
393         if (JavaVersion.JAVA_1_6.isOlderOrEqualTo(javacVersion)) {
394             // now add jdk 1.6 annotation processing related parameters
395 
396             if (config.getGeneratedSourcesDirectory() != null) {
397                 config.getGeneratedSourcesDirectory().mkdirs();
398                 args.add("-s");
399                 args.add(config.getGeneratedSourcesDirectory().getAbsolutePath());
400             }
401             if (config.getProc() != null) {
402                 args.add("-proc:" + config.getProc());
403             }
404             if (config.getAnnotationProcessors() != null) {
405                 args.add("-processor");
406                 String[] procs = config.getAnnotationProcessors();
407                 StringBuilder buffer = new StringBuilder();
408                 for (int i = 0; i < procs.length; i++) {
409                     if (i > 0) {
410                         buffer.append(",");
411                     }
412                     buffer.append(procs[i]);
413                 }
414                 args.add(buffer.toString());
415             }
416             if (config.getProcessorPathEntries() != null
417                     && !config.getProcessorPathEntries().isEmpty()) {
418                 args.add("-processorpath");
419                 args.add(getPathString(config.getProcessorPathEntries()));
420             }
421             if (config.getProcessorModulePathEntries() != null
422                     && !config.getProcessorModulePathEntries().isEmpty()) {
423                 args.add("--processor-module-path");
424                 args.add(getPathString(config.getProcessorModulePathEntries()));
425             }
426         }
427 
428         if (config.isOptimize()) {
429             args.add("-O");
430         }
431 
432         if (config.isDebug()) {
433             if (StringUtils.isNotEmpty(config.getDebugLevel())) {
434                 args.add("-g:" + config.getDebugLevel());
435             } else {
436                 args.add("-g");
437             }
438         }
439 
440         if (config.isVerbose()) {
441             args.add("-verbose");
442         }
443 
444         if (JavaVersion.JAVA_1_8.isOlderOrEqualTo(javacVersion) && config.isParameters()) {
445             args.add("-parameters");
446         }
447 
448         if (config.isEnablePreview()) {
449             args.add("--enable-preview");
450         }
451 
452         if (config.getImplicitOption() != null) {
453             args.add("-implicit:" + config.getImplicitOption());
454         }
455 
456         if (config.isShowDeprecation()) {
457             args.add("-deprecation");
458 
459             // This is required to actually display the deprecation messages
460             config.setShowWarnings(true);
461         }
462 
463         if (!config.isShowWarnings()) {
464             args.add("-nowarn");
465         } else {
466             String warnings = config.getWarnings();
467             if (config.isShowLint()) {
468                 if (config.isShowWarnings() && StringUtils.isNotEmpty(warnings)) {
469                     args.add("-Xlint:" + warnings);
470                 } else {
471                     args.add("-Xlint");
472                 }
473             }
474         }
475 
476         if (config.isFailOnWarning()) {
477             args.add("-Werror");
478         }
479 
480         if (JavaVersion.JAVA_9.isOlderOrEqualTo(javacVersion) && !StringUtils.isEmpty(config.getReleaseVersion())) {
481             args.add("--release");
482             args.add(config.getReleaseVersion());
483         } else {
484             // TODO: this could be much improved
485             if (StringUtils.isEmpty(config.getTargetVersion())) {
486                 // Required, or it defaults to the target of your JDK (eg 1.5)
487                 args.add("-target");
488                 args.add("1.1");
489             } else {
490                 args.add("-target");
491                 args.add(config.getTargetVersion());
492             }
493 
494             if (JavaVersion.JAVA_1_4.isOlderOrEqualTo(javacVersion) && StringUtils.isEmpty(config.getSourceVersion())) {
495                 // If omitted, later JDKs complain about a 1.1 target
496                 args.add("-source");
497                 args.add("1.3");
498             } else if (JavaVersion.JAVA_1_4.isOlderOrEqualTo(javacVersion)) {
499                 args.add("-source");
500                 args.add(config.getSourceVersion());
501             }
502         }
503 
504         if (JavaVersion.JAVA_1_4.isOlderOrEqualTo(javacVersion) && !StringUtils.isEmpty(config.getSourceEncoding())) {
505             args.add("-encoding");
506             args.add(config.getSourceEncoding());
507         }
508 
509         if (!StringUtils.isEmpty(config.getModuleVersion())) {
510             args.add("--module-version");
511             args.add(config.getModuleVersion());
512         }
513 
514         for (Map.Entry<String, String> entry : config.getCustomCompilerArgumentsEntries()) {
515             String key = entry.getKey();
516 
517             if (StringUtils.isEmpty(key) || key.startsWith("-J")) {
518                 continue;
519             }
520 
521             args.add(key);
522             String value = entry.getValue();
523             if (StringUtils.isEmpty(value)) {
524                 continue;
525             }
526             args.add(value);
527         }
528 
529         if (!config.isFork() && !args.contains("-XDuseUnsharedTable=false")) {
530             args.add("-XDuseUnsharedTable=true");
531         }
532 
533         return args.toArray(new String[0]);
534     }

572      * Compile the java sources in a external process, calling an external executable,
573      * like javac.
574      *
575      * @param config     compiler configuration
576      * @param executable name of the executable to launch
577      * @param args       arguments for the executable launched
578      * @return a CompilerResult object encapsulating the result of the compilation and any compiler messages
579      * @throws CompilerException
580      */
581     protected CompilerResult compileOutOfProcess(CompilerConfiguration config, String executable, String[] args)
582             throws CompilerException {
583         Commandline cli = new Commandline();
584 
585         cli.setWorkingDirectory(config.getWorkingDirectory().getAbsolutePath());
586         cli.setExecutable(executable);
587 
588         try {
589             File argumentsFile =
590                     createFileWithArguments(args, config.getBuildDirectory().getAbsolutePath());
591             cli.addArguments(
592                     new String[] {"@" + argumentsFile.getCanonicalPath().replace(File.separatorChar, '/')});
593 
594             if (!StringUtils.isEmpty(config.getMaxmem())) {
595                 cli.addArguments(new String[] {"-J-Xmx" + config.getMaxmem()});
596             }
597             if (!StringUtils.isEmpty(config.getMeminitial())) {
598                 cli.addArguments(new String[] {"-J-Xms" + config.getMeminitial()});
599             }
600 
601             for (String key : config.getCustomCompilerArgumentsAsMap().keySet()) {
602                 if (StringUtils.isNotEmpty(key) && key.startsWith("-J")) {
603                     cli.addArguments(new String[] {key});
604                 }
605             }
606         } catch (IOException e) {
607             throw new CompilerException("Error creating file with javac arguments", e);
608         }
609 
610         CommandLineUtils.StringStreamConsumer out = new CommandLineUtils.StringStreamConsumer();
611         int returnCode;
612         List<CompilerMessage> messages;
613 
614         if (getLog().isDebugEnabled()) {
615             String debugFileName = StringUtils.isEmpty(config.getDebugFileName()) ? "javac" : config.getDebugFileName();
616 
617             File commandLineFile = new File(
618                     config.getBuildDirectory(),
619                     StringUtils.trim(debugFileName) + "." + (Os.isFamily(Os.FAMILY_WINDOWS) ? "bat" : "sh"));
620             try {
621                 FileUtils.fileWrite(
622                         commandLineFile.getAbsolutePath(), cli.toString().replaceAll("'", ""));
623 
624                 if (!Os.isFamily(Os.FAMILY_WINDOWS)) {
625                     Runtime.getRuntime().exec(new String[] {"chmod", "a+x", commandLineFile.getAbsolutePath()});
626                 }
627             } catch (IOException e) {
628                 if (getLog().isWarnEnabled()) {
629                     getLog().warn("Unable to write '" + commandLineFile.getName() + "' debug script file", e);
630                 }
631             }
632         }
633 
634         try {
635             // TODO:
636             //   Is it really helpful to parse stdOut and stdErr as a single stream, instead of taking the chance to
637             //   draw extra information from the fact that normal javac output is written to stdOut, while warnings and
638             //   errors are written to stdErr? Of course, chronological correlation of messages would be more difficult
639             //   then, but basically, we are throwing away information here.
640             returnCode = CommandLineUtils.executeCommandLine(cli, out, out);
641 
642             if (getLog().isDebugEnabled()) {
643                 getLog().debug("Compiler output:{}{}", EOL, out.getOutput());
644             }
645 
646             messages = parseModernStream(returnCode, new BufferedReader(new StringReader(out.getOutput())));
647         } catch (CommandLineException | IOException e) {
648             throw new CompilerException("Error while executing the external compiler.", e);
649         }
650 
651         boolean success = returnCode == 0;
652         return new CompilerResult(success, messages);
653     }

710     // Match ~95% of existing JDK exception name patterns (last checked for JDK 21)
711     private static final Pattern STACK_TRACE_FIRST_LINE = Pattern.compile("^(?:[\\w+.-]+\\.)[\\w$]*?(?:"
712             + "Exception|Error|Throwable|Failure|Result|Abort|Fault|ThreadDeath|Overflow|Warning|"
713             + "NotSupported|NotFound|BadArgs|BadClassFile|Illegal|Invalid|Unexpected|Unchecked|Unmatched\\w+"
714             + ").*$");

716     // Match exception causes, existing and omitted stack trace elements
717     private static final Pattern STACK_TRACE_OTHER_LINE =
718             Pattern.compile("^(?:Caused by:\\s.*|\\s*at .*|\\s*\\.\\.\\.\\s\\d+\\smore)$");

720     /**
721      * Parse the compiler output into a list of compiler messages
722      *
723      * @param exitCode javac exit code (0 on success, non-zero otherwise)
724      * @param input    compiler output (stdOut and stdErr merged into input stream)
725      * @return list of {@link CompilerMessage} objects
726      * @throws IOException if there is a problem reading from the input reader
727      */
728     static List<CompilerMessage> parseModernStream(int exitCode, BufferedReader input) throws IOException {
729         List<CompilerMessage> errors = new ArrayList<>();
730         String line;
731         StringBuilder buffer = new StringBuilder();
732         boolean hasPointer = false;
733         int stackTraceLineCount = 0;
734 
735         while ((line = input.readLine()) != null) {
736             if (stackTraceLineCount == 0 && STACK_TRACE_FIRST_LINE.matcher(line).matches()
737                     || STACK_TRACE_OTHER_LINE.matcher(line).matches()) {
738                 stackTraceLineCount++;
739             } else {
740                 stackTraceLineCount = 0;
741             }
742 
743             // new error block?
744             if (!line.startsWith(" ") && hasPointer) {
745                 // add the error bean
746                 errors.add(parseModernError(exitCode, buffer.toString()));
747                 // reset for next error block
748                 buffer = new StringBuilder(); // this is quicker than clearing it
749                 hasPointer = false;
750             }
751 
752             if (buffer.length() == 0) {
753                 // try to classify output line by type (error, warning etc.)
754                 // TODO: there should be a better way to parse these
755                 if (isError(line)) {
756                     errors.add(new CompilerMessage(line, ERROR));
757                 } else if (isWarning(line)) {
758                     errors.add(new CompilerMessage(line, WARNING));
759                 } else if (isNote(line)) {
760                     // skip, JDK telling us deprecated APIs are used but -Xlint:deprecation isn't set
761                 } else if (isMisc(line)) {
762                     // verbose output was set
763                     errors.add(new CompilerMessage(line, CompilerMessage.Kind.OTHER));
764                 } else {
765                     // add first unclassified line to buffer
766                     buffer.append(line).append(EOL);
767                 }
768             } else {
769                 // add next unclassified line to buffer
770                 buffer.append(line).append(EOL);
771             }
772 
773             if (line.endsWith("^")) {
774                 hasPointer = true;
775             }
776         }
777 
778         String bufferContent = buffer.toString();
779         if (bufferContent.isEmpty()) {
780             return errors;
781         }
782 
783         // javac output not detected by other parsing
784         // maybe better to ignore only the summary and mark the rest as error
785         String cleanedUpMessage;
786         if ((cleanedUpMessage = getJavacGenericError(bufferContent)) != null
787                 || (cleanedUpMessage = getBootLayerInitError(bufferContent)) != null
788                 || (cleanedUpMessage = getVMInitError(bufferContent)) != null
789                 || (cleanedUpMessage = getFileABugError(bufferContent)) != null
790                 || (cleanedUpMessage = getAnnotationProcessingError(bufferContent)) != null
791                 || (cleanedUpMessage = getSystemOutOfResourcesError(bufferContent)) != null
792                 || (cleanedUpMessage = getIOError(bufferContent)) != null
793                 || (cleanedUpMessage = getPluginError(bufferContent)) != null) {
794             errors.add(new CompilerMessage(cleanedUpMessage, ERROR));
795         } else if (hasPointer) {
796             // A compiler message remains in buffer at end of parse stream
797             errors.add(parseModernError(exitCode, bufferContent));
798         } else if (stackTraceLineCount > 0) {
799             // Extract stack trace from end of buffer
800             String[] lines = bufferContent.split("\\R");
801             int linesTotal = lines.length;
802             buffer = new StringBuilder();
803             int firstLine = linesTotal - stackTraceLineCount;
804             for (int i = firstLine; i < linesTotal; i++) {
805                 buffer.append(lines[i]).append(EOL);
806             }
807             errors.add(new CompilerMessage(buffer.toString(), ERROR));
808         }
809         // TODO: Add something like this? Check if it creates more value or more unnecessary log output in general.
810         // else {
811         //     // Fall-back, if still no error or stack trace was recognised
812         //     errors.add(new CompilerMessage(bufferContent, exitCode == 0 ? OTHER : ERROR));
813         // }
814 
815         return errors;
816     }

875     /**
876      * Identify and return a known javac error message prefix and all subsequent text - usually a stack trace - from a
877      * javac log output buffer.
878      *
879      * @param text     log buffer to search for a javac error message stack trace
880      * @param prefixes array of strings in Java properties format, e.g. {@code "some error with line feed\nand parameter
881      *                 placeholders {0} and {1}"} in multiple locales (hence the array). For the search, the
882      *                 placeholders may be represented by any text in the log buffer.
883      * @return if found, the error message + all subsequent text, otherwise {@code null}
884      */
885     static String getTextStartingWithPrefix(String text, String[] prefixes) {
886         // Implementation note: The properties format with placeholders  makes it easy to just copy & paste values from
887         // the JDK compared to having to convert them to regular expressions with ".*" instead of "{0}" and quote
888         // special regex characters. This makes the implementation of this method more complex and potentially a bit
889         // slower, but hopefully is worth the effort for the convenience of future developers maintaining this class.
890 
891         // Normalise line feeds to the UNIX format found in JDK multi-line messages in properties files
892         text = text.replaceAll("\\R", "\n");
893 
894         // Search text for given error message prefixes/headers, until the first match is found
895         for (String prefix : prefixes) {
896             // Split properties message along placeholders like "{0}", "{1}" etc.
897             String[] prefixParts = prefix.split("\\{\\d+\\}");
898             for (int i = 0; i < prefixParts.length; i++) {
899                 // Make sure to treat split sections as literal text in search regex by enclosing them in "\Q" and "\E".
900                 // See https://docs.oracle.com/javase/8/docs/api/java/util/regex/Pattern.html, search for "Quotation".
901                 prefixParts[i] = "\\Q" + prefixParts[i] + "\\E";
902             }
903             // Join message parts, replacing properties placeholders by ".*" regex ones
904             prefix = String.join(".*?", prefixParts);
905             // Find prefix + subsequent text in Pattern.DOTALL mode, represented in regex as "(?s)".
906             // This matches across line break boundaries.
907             Matcher matcher = Pattern.compile("(?s).*(" + prefix + ".*)").matcher(text);
908             if (matcher.matches()) {
909                 // Match -> cut off text before header and replace UNIX line breaks by platform ones again
910                 return matcher.replaceFirst("$1").replaceAll("\n", EOL);
911             }
912         }
913 
914         // No match
915         return null;
916     }

918     /**
919      * Construct a compiler message object from a compiler output line
920      *
921      * @param exitCode javac exit code
922      * @param error    compiler output line
923      * @return compiler message object
924      */
925     static CompilerMessage parseModernError(int exitCode, String error) {
926         final StringTokenizer tokens = new StringTokenizer(error, ":");
927         CompilerMessage.Kind messageKind = exitCode == 0 ? WARNING : ERROR;
928 
929         try {
930             // With Java 6 error output lines from the compiler got longer. For backward compatibility
931             // and the time being, we eat up all (if any) tokens up to the erroneous file and source
932             // line indicator tokens.
933 
934             boolean tokenIsAnInteger;
935             StringBuilder file = null;
936             String currentToken = null;
937 
938             do {
939                 if (currentToken != null) {
940                     if (file == null) {
941                         file = new StringBuilder(currentToken);
942                     } else {
943                         file.append(':').append(currentToken);
944                     }
945                 }
946 
947                 currentToken = tokens.nextToken();
948                 // Probably the only backward compatible means of checking if a string is an integer.
949                 tokenIsAnInteger = true;
950 
951                 try {
952                     Integer.parseInt(currentToken);
953                 } catch (NumberFormatException e) {
954                     tokenIsAnInteger = false;
955                 }
956             } while (!tokenIsAnInteger);
957 
958             final String lineIndicator = currentToken;
959             final int startOfFileName = Objects.requireNonNull(file).toString().lastIndexOf(']');
960             if (startOfFileName > -1) {
961                 file = new StringBuilder(file.substring(startOfFileName + 1 + EOL.length()));
962             }
963 
964             final int line = Integer.parseInt(lineIndicator);
965             final StringBuilder msgBuffer = new StringBuilder();
966             String msg = tokens.nextToken(EOL).substring(2);
967 
968             // Remove "error: " and "warning: " prefixes
969             String prefix;
970             if ((prefix = getErrorPrefix(msg)) != null) {
971                 messageKind = ERROR;
972                 msg = msg.substring(prefix.length());
973             } else if ((prefix = getWarningPrefix(msg)) != null) {
974                 messageKind = WARNING;
975                 msg = msg.substring(prefix.length());
976             }
977             msgBuffer.append(msg).append(EOL);
978 
979             String context = tokens.nextToken(EOL);
980             String pointer = null;
981 
982             do {
983                 final String msgLine = tokens.nextToken(EOL);
984                 if (pointer != null) {
985                     msgBuffer.append(msgLine);
986                     msgBuffer.append(EOL);
987                 } else if (msgLine.endsWith("^")) {
988                     pointer = msgLine;
989                 } else {
990                     msgBuffer.append(context);
991                     msgBuffer.append(EOL);
992                     context = msgLine;
993                 }
994             } while (tokens.hasMoreTokens());
995 
996             msgBuffer.append(EOL);
997 
998             final String message = msgBuffer.toString();
999             final int startcolumn = Objects.requireNonNull(pointer).indexOf("^");
1000             int endcolumn = (context == null) ? startcolumn : context.indexOf(" ", startcolumn);
1001             if (endcolumn == -1) {
1002                 endcolumn = Objects.requireNonNull(context).length();
1003             }
1004 
1005             return new CompilerMessage(
1006                     file.toString(), messageKind, line, startcolumn, line, endcolumn, message.trim());
1007         } catch (NoSuchElementException e) {
1008             return new CompilerMessage("no more tokens - could not parse error message: " + error, messageKind);
1009         } catch (Exception e) {
1010             return new CompilerMessage("could not parse error message: " + error, messageKind);
1011         }
1012     }

1095     private static String getJavacExecutable() throws IOException {
1096         String javacCommand = "javac" + (Os.isFamily(Os.FAMILY_WINDOWS) ? ".exe" : "");
1097         String javaHome = System.getProperty("java.home");
1098         File javacExe;
1099 
1100         if (Os.isName("AIX")) {
1101             javacExe = new File(javaHome + File.separator + ".." + File.separator + "sh", javacCommand);
1102         } else if (Os.isName("Mac OS X")) {
1103             javacExe = new File(javaHome + File.separator + "bin", javacCommand);
1104         } else {
1105             javacExe = new File(javaHome + File.separator + ".." + File.separator + "bin", javacCommand);
1106         }
1107 
1108         // ----------------------------------------------------------------------
1109         // Try to find javacExe from JAVA_HOME environment variable
1110         // ----------------------------------------------------------------------
1111         if (!javacExe.isFile()) {
1112             Properties env = CommandLineUtils.getSystemEnvVars();
1113             javaHome = env.getProperty("JAVA_HOME");
1114             if (StringUtils.isEmpty(javaHome)) {
1115                 throw new IOException("The environment variable JAVA_HOME is not correctly set.");
1116             }
1117             if (!new File(javaHome).isDirectory()) {
1118                 throw new IOException("The environment variable JAVA_HOME=" + javaHome
1119                         + " doesn't exist or is not a valid directory.");
1120             }
1121             javacExe = new File(env.getProperty("JAVA_HOME") + File.separator + "bin", javacCommand);
1122         }
1123 
1124         if (!javacExe.isFile()) {
1125             throw new IOException("The javadoc executable '" + javacExe
1126                     + "' doesn't exist or is not a file. Verify the JAVA_HOME environment variable.");
1127         }
1128 
1129         return javacExe.getAbsolutePath();
1130     }

1139     /**
1140      * Find the main class of JavaC. Return the same class for subsequent calls.
1141      *
1142      * @return the non-null class.
1143      * @throws CompilerException if the class has not been found.
1144      */
1145     private Class<?> getJavacClass(CompilerConfiguration compilerConfiguration) throws CompilerException {
1146         Class<?> c;
1147         switch (compilerConfiguration.getCompilerReuseStrategy()) {
1148             case AlwaysNew:
1149                 return createJavacClass();
1150             case ReuseCreated:
1151                 c = javacClasses.poll();
1152                 if (c == null) {
1153                     c = createJavacClass();
1154                 }
1155                 return c;
1156             case ReuseSame:
1157             default:
1158                 c = javacClass;
1159                 if (c == null) {
1160                     synchronized (this) {
1161                         c = javacClass;
1162                         if (c == null) {
1163                             javacClass = c = createJavacClass();
1164                         }
1165                     }
1166                 }
1167                 return c;
1168         }
1169     }

1171     /**
1172      * Helper method for create Javac class
1173      */
1174     protected Class<?> createJavacClass() throws CompilerException {
1175         try {
1176             // look whether JavaC is on Maven's classpath
1177             // return Class.forName( JavacCompiler.JAVAC_CLASSNAME, true, JavacCompiler.class.getClassLoader() );
1178             return JavacCompiler.class.getClassLoader().loadClass(JavacCompiler.JAVAC_CLASSNAME);
1179         } catch (ClassNotFoundException ex) {
1180             // ok
1181         }
1182 
1183         final File toolsJar = new File(System.getProperty("java.home"), "../lib/tools.jar");
1184         if (!toolsJar.exists()) {
1185             throw new CompilerException("tools.jar not found: " + toolsJar);
1186         }
1187 
1188         try {
1189             // Combined classloader with no parent/child relationship, so classes in our classloader
1190             // can reference classes in tools.jar
1191             URL[] originalUrls = ((URLClassLoader) JavacCompiler.class.getClassLoader()).getURLs();
1192             URL[] urls = new URL[originalUrls.length + 1];
1193             urls[0] = toolsJar.toURI().toURL();
1194             System.arraycopy(originalUrls, 0, urls, 1, originalUrls.length);
1195             ClassLoader javacClassLoader = new URLClassLoader(urls);
1196 
1197             final Thread thread = Thread.currentThread();
1198             final ClassLoader contextClassLoader = thread.getContextClassLoader();
1199             thread.setContextClassLoader(javacClassLoader);
1200             try {
1201                 // return Class.forName( JavacCompiler.JAVAC_CLASSNAME, true, javacClassLoader );
1202                 return javacClassLoader.loadClass(JavacCompiler.JAVAC_CLASSNAME);
1203             } finally {
1204                 thread.setContextClassLoader(contextClassLoader);
1205             }
1206         } catch (MalformedURLException ex) {
1207             throw new CompilerException(
1208                     "Could not convert the file reference to tools.jar to a URL, path to tools.jar: '"
1209                             + toolsJar.getAbsolutePath() + "'.",
1210                     ex);
1211         } catch (ClassNotFoundException ex) {
1212             throw new CompilerException(
1213                     "Unable to locate the Javac Compiler in:" + EOL + "  " + toolsJar + EOL
1214                             + "Please ensure you are using JDK 1.4 or above and" + EOL
1215                             + "not a JRE (the com.sun.tools.javac.Main class is required)." + EOL
1216                             + "In most cases you can change the location of your Java" + EOL
1217                             + "installation by setting the JAVA_HOME environment variable.",
1218                     ex);
1219         }
1220     }
