131     @Override
132     public CompilerResult performCompile(CompilerConfiguration config) throws CompilerException {
133         File destinationDir = new File(config.getOutputLocation());
134 
135         if (!destinationDir.exists()) {
136             destinationDir.mkdirs();
137         }
138 
139         String[] sourceFiles = getSourceFiles(config);
140 
141         if ((sourceFiles == null) || (sourceFiles.length == 0)) {
142             return new CompilerResult();
143         }
144 
145         logCompiling(sourceFiles, config);
146 
147         String[] args = buildCompilerArguments(config, sourceFiles);
148 
149         CompilerResult result;
150 
151         if (config.isFork()) {
152             String executable = config.getExecutable();
153 
154             if (StringUtils.isEmpty(executable)) {
155                 try {
156                     executable = getJavacExecutable();
157                 } catch (IOException e) {
158                     if (getLog().isWarnEnabled()) {
159                         getLog().warn("Unable to autodetect 'javac' path, using 'javac' from the environment.");
160                     }
161                     executable = "javac";
162                 }
163             }
164 
165             result = compileOutOfProcess(config, executable, args);
166         } else {
167             if (isJava16() && !config.isForceJavacCompilerUse()) {
168                 // use fqcn to prevent loading of the class on 1.5 environment !
169                 result = inProcessCompiler().compileInProcess(args, config, sourceFiles);
170             } else {
171                 result = compileInProcess(args, config);
172             }
173         }
174 
175         return result;
176     }

195     public static String[] buildCompilerArguments(CompilerConfiguration config, String[] sourceFiles) {
196         List<String> args = new ArrayList<>();
197 
198         // ----------------------------------------------------------------------
199         // Set output
200         // ----------------------------------------------------------------------
201 
202         File destinationDir = new File(config.getOutputLocation());
203 
204         args.add("-d");
205 
206         args.add(destinationDir.getAbsolutePath());
207 
208         // ----------------------------------------------------------------------
209         // Set the class and source paths
210         // ----------------------------------------------------------------------
211 
212         List<String> classpathEntries = config.getClasspathEntries();
213         if (classpathEntries != null && !classpathEntries.isEmpty()) {
214             args.add("-classpath");
215 
216             args.add(getPathString(classpathEntries));
217         }
218 
219         List<String> modulepathEntries = config.getModulepathEntries();
220         if (modulepathEntries != null && !modulepathEntries.isEmpty()) {
221             args.add("--module-path");
222 
223             args.add(getPathString(modulepathEntries));
224         }
225 
226         List<String> sourceLocations = config.getSourceLocations();
227         if (sourceLocations != null && !sourceLocations.isEmpty()) {
228             // always pass source path, even if sourceFiles are declared,
229             // needed for jsr269 annotation processing, see MCOMPILER-98
230             args.add("-sourcepath");
231 
232             args.add(getPathString(sourceLocations));
233         }
234         if (!isJava16() || config.isForceJavacCompilerUse() || config.isFork()) {
235             args.addAll(Arrays.asList(sourceFiles));
236         }
237 
238         if (!isPreJava16(config)) {
239             // now add jdk 1.6 annotation processing related parameters
240 
241             if (config.getGeneratedSourcesDirectory() != null) {
242                 config.getGeneratedSourcesDirectory().mkdirs();
243 
244                 args.add("-s");
245                 args.add(config.getGeneratedSourcesDirectory().getAbsolutePath());
246             }
247             if (config.getProc() != null) {
248                 args.add("-proc:" + config.getProc());
249             }
250             if (config.getAnnotationProcessors() != null) {
251                 args.add("-processor");
252                 String[] procs = config.getAnnotationProcessors();
253                 StringBuilder buffer = new StringBuilder();
254                 for (int i = 0; i < procs.length; i++) {
255                     if (i > 0) {
256                         buffer.append(",");
257                     }
258 
259                     buffer.append(procs[i]);
260                 }
261                 args.add(buffer.toString());
262             }
263             if (config.getProcessorPathEntries() != null
264                     && !config.getProcessorPathEntries().isEmpty()) {
265                 args.add("-processorpath");
266                 args.add(getPathString(config.getProcessorPathEntries()));
267             }
268             if (config.getProcessorModulePathEntries() != null
269                     && !config.getProcessorModulePathEntries().isEmpty()) {
270                 args.add("--processor-module-path");
271                 args.add(getPathString(config.getProcessorModulePathEntries()));
272             }
273         }
274 
275         if (config.isOptimize()) {
276             args.add("-O");
277         }
278 
279         if (config.isDebug()) {
280             if (StringUtils.isNotEmpty(config.getDebugLevel())) {
281                 args.add("-g:" + config.getDebugLevel());
282             } else {
283                 args.add("-g");
284             }
285         }
286 
287         if (config.isVerbose()) {
288             args.add("-verbose");
289         }
290 
291         if (!isPreJava18(config) && config.isParameters()) {
292             args.add("-parameters");
293         }
294 
295         if (config.isEnablePreview()) {
296             args.add("--enable-preview");
297         }
298 
299         if (config.getImplicitOption() != null) {
300             args.add("-implicit:" + config.getImplicitOption());
301         }
302 
303         if (config.isShowDeprecation()) {
304             args.add("-deprecation");
305 
306             // This is required to actually display the deprecation messages
307             config.setShowWarnings(true);
308         }
309 
310         if (!config.isShowWarnings()) {
311             args.add("-nowarn");
312         } else {
313             String warnings = config.getWarnings();
314             if (config.isShowLint()) {
315                 if (config.isShowWarnings() && StringUtils.isNotEmpty(warnings)) {
316                     args.add("-Xlint:" + warnings);
317                 } else {
318                     args.add("-Xlint");
319                 }
320             }
321         }
322 
323         if (config.isFailOnWarning()) {
324             args.add("-Werror");
325         }
326 
327         if (!StringUtils.isEmpty(config.getReleaseVersion())) {
328             args.add("--release");
329             args.add(config.getReleaseVersion());
330         } else {
331             // TODO: this could be much improved
332             if (StringUtils.isEmpty(config.getTargetVersion())) {
333                 // Required, or it defaults to the target of your JDK (eg 1.5)
334                 args.add("-target");
335                 args.add("1.1");
336             } else {
337                 args.add("-target");
338                 args.add(config.getTargetVersion());
339             }
340 
341             if (!suppressSource(config) && StringUtils.isEmpty(config.getSourceVersion())) {
342                 // If omitted, later JDKs complain about a 1.1 target
343                 args.add("-source");
344                 args.add("1.3");
345             } else if (!suppressSource(config)) {
346                 args.add("-source");
347                 args.add(config.getSourceVersion());
348             }
349         }
350 
351         if (!suppressEncoding(config) && !StringUtils.isEmpty(config.getSourceEncoding())) {
352             args.add("-encoding");
353             args.add(config.getSourceEncoding());
354         }
355 
356         if (!StringUtils.isEmpty(config.getModuleVersion())) {
357             args.add("--module-version");
358             args.add(config.getModuleVersion());
359         }
360 
361         for (Map.Entry<String, String> entry : config.getCustomCompilerArgumentsEntries()) {
362             String key = entry.getKey();
363 
364             if (StringUtils.isEmpty(key) || key.startsWith("-J")) {
365                 continue;
366             }
367 
368             args.add(key);
369 
370             String value = entry.getValue();
371 
372             if (StringUtils.isEmpty(value)) {
373                 continue;
374             }
375 
376             args.add(value);
377         }
378 
379         if (!config.isFork()) {
380             args.add("-XDuseUnsharedTable=true");
381         }
382 
383         return args.toArray(new String[0]);
384     }

498     /**
499      * Compile the java sources in a external process, calling an external executable,
500      * like javac.
501      *
502      * @param config     compiler configuration
503      * @param executable name of the executable to launch
504      * @param args       arguments for the executable launched
505      * @return a CompilerResult object encapsulating the result of the compilation and any compiler messages
506      * @throws CompilerException
507      */
508     protected CompilerResult compileOutOfProcess(CompilerConfiguration config, String executable, String[] args)
509             throws CompilerException {
510         Commandline cli = new Commandline();
511 
512         cli.setWorkingDirectory(config.getWorkingDirectory().getAbsolutePath());
513 
514         cli.setExecutable(executable);
515 
516         try {
517             File argumentsFile =
518                     createFileWithArguments(args, config.getBuildDirectory().getAbsolutePath());
519             cli.addArguments(
520                     new String[] {"@" + argumentsFile.getCanonicalPath().replace(File.separatorChar, '/')});
521 
522             if (!StringUtils.isEmpty(config.getMaxmem())) {
523                 cli.addArguments(new String[] {"-J-Xmx" + config.getMaxmem()});
524             }
525 
526             if (!StringUtils.isEmpty(config.getMeminitial())) {
527                 cli.addArguments(new String[] {"-J-Xms" + config.getMeminitial()});
528             }
529 
530             for (String key : config.getCustomCompilerArgumentsAsMap().keySet()) {
531                 if (StringUtils.isNotEmpty(key) && key.startsWith("-J")) {
532                     cli.addArguments(new String[] {key});
533                 }
534             }
535         } catch (IOException e) {
536             throw new CompilerException("Error creating file with javac arguments", e);
537         }
538 
539         CommandLineUtils.StringStreamConsumer out = new CommandLineUtils.StringStreamConsumer();
540 
541         int returnCode;
542 
543         List<CompilerMessage> messages;
544 
545         if (getLog().isDebugEnabled()) {
546             String debugFileName = StringUtils.isEmpty(config.getDebugFileName()) ? "javac" : config.getDebugFileName();
547 
548             File commandLineFile = new File(
549                     config.getBuildDirectory(),
550                     StringUtils.trim(debugFileName) + "." + (Os.isFamily(Os.FAMILY_WINDOWS) ? "bat" : "sh"));
551             try {
552                 FileUtils.fileWrite(
553                         commandLineFile.getAbsolutePath(), cli.toString().replaceAll("'", ""));
554 
555                 if (!Os.isFamily(Os.FAMILY_WINDOWS)) {
556                     Runtime.getRuntime().exec(new String[] {"chmod", "a+x", commandLineFile.getAbsolutePath()});
557                 }
558             } catch (IOException e) {
559                 if (getLog().isWarnEnabled()) {
560                     getLog().warn("Unable to write '" + commandLineFile.getName() + "' debug script file", e);
561                 }
562             }
563         }
564 
565         try {
566             returnCode = CommandLineUtils.executeCommandLine(cli, out, out);
567 
568             messages = parseModernStream(returnCode, new BufferedReader(new StringReader(out.getOutput())));
569         } catch (CommandLineException | IOException e) {
570             throw new CompilerException("Error while executing the external compiler.", e);
571         }
572 
573         boolean success = returnCode == 0;
574         return new CompilerResult(success, messages);
575     }

577     /**
578      * Compile the java sources in the current JVM, without calling an external executable,
579      * using <code>com.sun.tools.javac.Main</code> class
580      *
581      * @param args   arguments for the compiler as they would be used in the command line javac
582      * @param config compiler configuration
583      * @return a CompilerResult object encapsulating the result of the compilation and any compiler messages
584      * @throws CompilerException
585      */
586     CompilerResult compileInProcess(String[] args, CompilerConfiguration config) throws CompilerException {
587         final Class<?> javacClass = getJavacClass(config);
588         final Thread thread = Thread.currentThread();
589         final ClassLoader contextClassLoader = thread.getContextClassLoader();
590         thread.setContextClassLoader(javacClass.getClassLoader());
591         if (getLog().isDebugEnabled()) {
592             getLog().debug("ttcl changed run compileInProcessWithProperClassloader");
593         }
594         try {
595             return compileInProcessWithProperClassloader(javacClass, args);
596         } finally {
597             releaseJavaccClass(javacClass, config);
598             thread.setContextClassLoader(contextClassLoader);
599         }
600     }

607     /**
608      * Helper method for compileInProcess()
609      */
610     private static CompilerResult compileInProcess0(Class<?> javacClass, String[] args) throws CompilerException {
611         StringWriter out = new StringWriter();
612 
613         Integer ok;
614 
615         List<CompilerMessage> messages;
616 
617         try {
618             Method compile = javacClass.getMethod("compile", new Class[] {String[].class, PrintWriter.class});
619 
620             ok = (Integer) compile.invoke(null, new Object[] {args, new PrintWriter(out)});
621 
622             messages = parseModernStream(ok, new BufferedReader(new StringReader(out.toString())));
623         } catch (NoSuchMethodException | IOException | InvocationTargetException | IllegalAccessException e) {
624             throw new CompilerException("Error while executing the compiler.", e);
625         }
626 
627         boolean success = ok == 0;
628         return new CompilerResult(success, messages);
629     }

641     /**
642      * Parse the output from the compiler into a list of CompilerMessage objects
643      *
644      * @param exitCode The exit code of javac.
645      * @param input    The output of the compiler
646      * @return List of CompilerMessage objects
647      * @throws IOException
648      */
649     static List<CompilerMessage> parseModernStream(int exitCode, BufferedReader input) throws IOException {
650         List<CompilerMessage> errors = new ArrayList<>();
651 
652         String line;
653 
654         StringBuilder buffer = new StringBuilder();
655 
656         boolean hasPointer = false;
657         int stackTraceLineCount = 0;
658 
659         while (true) {
660             line = input.readLine();
661 
662             if (line == null) {
663                 // javac output not detected by other parsing
664                 // maybe better to ignore only the summary and mark the rest as error
665                 String bufferAsString = buffer.toString();
666                 if (buffer.length() > 0) {
667                     if (bufferAsString.startsWith("javac:")) {
668                         errors.add(new CompilerMessage(bufferAsString, CompilerMessage.Kind.ERROR));
669                     } else if (bufferAsString.startsWith("Error occurred during initialization of boot layer")) {
670                         errors.add(new CompilerMessage(bufferAsString, CompilerMessage.Kind.OTHER));
671                     } else if (hasPointer) {
672                         // A compiler message remains in buffer at end of parse stream
673                         errors.add(parseModernError(exitCode, bufferAsString));
674                     } else if (stackTraceLineCount > 0) {
675                         // Extract stack trace from end of buffer
676                         String[] lines = bufferAsString.split("\\R");
677                         int linesTotal = lines.length;
678                         buffer = new StringBuilder();
679                         int firstLine = linesTotal - stackTraceLineCount;
680 
681                         // Salvage Javac localized message 'javac.msg.bug' ("An exception has occurred in the
682                         // compiler ... Please file a bug")
683                         if (firstLine > 0) {
684                             final String lineBeforeStackTrace = lines[firstLine - 1];
685                             // One of those two URL substrings should always appear, without regard to JVM locale.
686                             // TODO: Update, if the URL changes, last checked for JDK 21.
687                             if (lineBeforeStackTrace.contains("java.sun.com/webapps/bugreport")
688                                     || lineBeforeStackTrace.contains("bugreport.java.com")) {
689                                 firstLine--;
690                             }
691                         }
692 
693                         // Note: For message 'javac.msg.proc.annotation.uncaught.exception' ("An annotation processor
694                         // threw an uncaught exception"), there is no locale-independent substring, and the header is
695                         // also multi-line. It was discarded in the removed method 'parseAnnotationProcessorStream',
696                         // and we continue to do so.
697 
698                         for (int i = firstLine; i < linesTotal; i++) {
699                             buffer.append(lines[i]).append(EOL);
700                         }
701                         errors.add(new CompilerMessage(buffer.toString(), CompilerMessage.Kind.ERROR));
702                     }
703                 }
704                 return errors;
705             }
706 
707             if (stackTraceLineCount == 0 && STACK_TRACE_FIRST_LINE.matcher(line).matches()
708                     || STACK_TRACE_OTHER_LINE.matcher(line).matches()) {
709                 stackTraceLineCount++;
710             } else {
711                 stackTraceLineCount = 0;
712             }
713 
714             // new error block?
715             if (!line.startsWith(" ") && hasPointer) {
716                 // add the error bean
717                 errors.add(parseModernError(exitCode, buffer.toString()));
718 
719                 // reset for next error block
720                 buffer = new StringBuilder(); // this is quicker than clearing it
721 
722                 hasPointer = false;
723             }
724 
725             // TODO: there should be a better way to parse these
726             if ((buffer.length() == 0) && line.startsWith("error: ")) {
727                 errors.add(new CompilerMessage(line, CompilerMessage.Kind.ERROR));
728             } else if ((buffer.length() == 0) && line.startsWith("warning: ")) {
729                 errors.add(new CompilerMessage(line, CompilerMessage.Kind.WARNING));
730             } else if ((buffer.length() == 0) && isNote(line)) {
731                 // skip, JDK 1.5 telling us deprecated APIs are used but -Xlint:deprecation isn't set
732             } else if ((buffer.length() == 0) && isMisc(line)) {
733                 // verbose output was set
734                 errors.add(new CompilerMessage(line, CompilerMessage.Kind.OTHER));
735             } else {
736                 buffer.append(line);
737 
738                 buffer.append(EOL);
739             }
740 
741             if (line.endsWith("^")) {
742                 hasPointer = true;
743             }
744         }
745     }

748         return startsWithPrefix(line, MISC_PREFIXES);

764     /**
765      * Construct a CompilerMessage object from a line of the compiler output
766      *
767      * @param exitCode The exit code from javac.
768      * @param error    output line from the compiler
769      * @return the CompilerMessage object
770      */
771     static CompilerMessage parseModernError(int exitCode, String error) {
772         final StringTokenizer tokens = new StringTokenizer(error, ":");
773 
774         boolean isError = exitCode != 0;
775 
776         try {
777             // With Java 6 error output lines from the compiler got longer. For backward compatibility
778             // .. and the time being, we eat up all (if any) tokens up to the erroneous file and source
779             // .. line indicator tokens.
780 
781             boolean tokenIsAnInteger;
782 
783             StringBuilder file = null;
784 
785             String currentToken = null;
786 
787             do {
788                 if (currentToken != null) {
789                     if (file == null) {
790                         file = new StringBuilder(currentToken);
791                     } else {
792                         file.append(':').append(currentToken);
793                     }
794                 }
795 
796                 currentToken = tokens.nextToken();
797 
798                 // Probably the only backward compatible means of checking if a string is an integer.
799 
800                 tokenIsAnInteger = true;
801 
802                 try {
803                     Integer.parseInt(currentToken);
804                 } catch (NumberFormatException e) {
805                     tokenIsAnInteger = false;
806                 }
807             } while (!tokenIsAnInteger);
808 
809             final String lineIndicator = currentToken;
810 
811             final int startOfFileName = file.toString().lastIndexOf(']');
812 
813             if (startOfFileName > -1) {
814                 file = new StringBuilder(file.substring(startOfFileName + 1 + EOL.length()));
815             }
816 
817             final int line = Integer.parseInt(lineIndicator);
818 
819             final StringBuilder msgBuffer = new StringBuilder();
820 
821             String msg = tokens.nextToken(EOL).substring(2);
822 
823             // Remove the 'warning: ' prefix
824             final String warnPrefix = getWarnPrefix(msg);
825             if (warnPrefix != null) {
826                 isError = false;
827                 msg = msg.substring(warnPrefix.length());
828             } else {
829                 isError = exitCode != 0;
830             }
831 
832             msgBuffer.append(msg);
833 
834             msgBuffer.append(EOL);
835 
836             String context = tokens.nextToken(EOL);
837 
838             String pointer = null;
839 
840             do {
841                 final String msgLine = tokens.nextToken(EOL);
842 
843                 if (pointer != null) {
844                     msgBuffer.append(msgLine);
845 
846                     msgBuffer.append(EOL);
847                 } else if (msgLine.endsWith("^")) {
848                     pointer = msgLine;
849                 } else {
850                     msgBuffer.append(context);
851 
852                     msgBuffer.append(EOL);
853 
854                     context = msgLine;
855                 }
856             } while (tokens.hasMoreTokens());
857 
858             msgBuffer.append(EOL);
859 
860             final String message = msgBuffer.toString();
861 
862             final int startcolumn = pointer.indexOf("^");
863 
864             int endcolumn = (context == null) ? startcolumn : context.indexOf(" ", startcolumn);
865 
866             if (endcolumn == -1) {
867                 endcolumn = context.length();
868             }
869 
870             return new CompilerMessage(file.toString(), isError, line, startcolumn, line, endcolumn, message.trim());
871         } catch (NoSuchElementException e) {
872             return new CompilerMessage("no more tokens - could not parse error message: " + error, isError);
873         } catch (Exception e) {
874             return new CompilerMessage("could not parse error message: " + error, isError);
875         }
876     }

878     private static String getWarnPrefix(String msg) {
879         for (String warningPrefix : WARNING_PREFIXES) {
880             if (msg.startsWith(warningPrefix)) {
881                 return warningPrefix;
882             }
883         }
884         return null;
885     }

887     /**
888      * put args into a temp file to be referenced using the @ option in javac command line
889      *
890      * @param args
891      * @return the temporary file wth the arguments
892      * @throws IOException
893      */
894     private File createFileWithArguments(String[] args, String outputDirectory) throws IOException {
895         PrintWriter writer = null;
896         try {
897             File tempFile;
898             if (getLog().isDebugEnabled()) {
899                 tempFile = File.createTempFile(JavacCompiler.class.getName(), "arguments", new File(outputDirectory));
900             } else {
901                 tempFile = File.createTempFile(JavacCompiler.class.getName(), "arguments");
902                 tempFile.deleteOnExit();
903             }
904 
905             writer = new PrintWriter(new FileWriter(tempFile));
906 
907             for (String arg : args) {
908                 String argValue = arg.replace(File.separatorChar, '/');
909 
910                 writer.write("\"" + argValue + "\"");
911 
912                 writer.println();
913             }
914 
915             writer.flush();
916 
917             return tempFile;
918 
919         } finally {
920             if (writer != null) {
921                 writer.close();
922             }
923         }
924     }

926     /**
927      * Get the path of the javac tool executable: try to find it depending the OS or the <code>java.home</code>
928      * system property or the <code>JAVA_HOME</code> environment variable.
929      *
930      * @return the path of the Javadoc tool
931      * @throws IOException if not found
932      */
933     private static String getJavacExecutable() throws IOException {
934         String javacCommand = "javac" + (Os.isFamily(Os.FAMILY_WINDOWS) ? ".exe" : "");
935 
936         String javaHome = System.getProperty("java.home");
937         File javacExe;
938         if (Os.isName("AIX")) {
939             javacExe = new File(javaHome + File.separator + ".." + File.separator + "sh", javacCommand);
940         } else if (Os.isName("Mac OS X")) {
941             javacExe = new File(javaHome + File.separator + "bin", javacCommand);
942         } else {
943             javacExe = new File(javaHome + File.separator + ".." + File.separator + "bin", javacCommand);
944         }
945 
946         // ----------------------------------------------------------------------
947         // Try to find javacExe from JAVA_HOME environment variable
948         // ----------------------------------------------------------------------
949         if (!javacExe.isFile()) {
950             Properties env = CommandLineUtils.getSystemEnvVars();
951             javaHome = env.getProperty("JAVA_HOME");
952             if (StringUtils.isEmpty(javaHome)) {
953                 throw new IOException("The environment variable JAVA_HOME is not correctly set.");
954             }
955             if (!new File(javaHome).isDirectory()) {
956                 throw new IOException("The environment variable JAVA_HOME=" + javaHome
957                         + " doesn't exist or is not a valid directory.");
958             }
959 
960             javacExe = new File(env.getProperty("JAVA_HOME") + File.separator + "bin", javacCommand);
961         }
962 
963         if (!javacExe.isFile()) {
964             throw new IOException("The javadoc executable '" + javacExe
965                     + "' doesn't exist or is not a file. Verify the JAVA_HOME environment variable.");
966         }
967 
968         return javacExe.getAbsolutePath();
969     }

978     /**
979      * Find the main class of JavaC. Return the same class for subsequent calls.
980      *
981      * @return the non-null class.
982      * @throws CompilerException if the class has not been found.
983      */
984     private Class<?> getJavacClass(CompilerConfiguration compilerConfiguration) throws CompilerException {
985         Class<?> c;
986         switch (compilerConfiguration.getCompilerReuseStrategy()) {
987             case AlwaysNew:
988                 return createJavacClass();
989             case ReuseCreated:
990                 c = javacClasses.poll();
991                 if (c == null) {
992                     c = createJavacClass();
993                 }
994                 return c;
995             case ReuseSame:
996             default:
997                 c = javacClass;
998                 if (c == null) {
999                     synchronized (this) {
1000                         c = javacClass;
1001                         if (c == null) {
1002                             javacClass = c = createJavacClass();
1003                         }
1004                     }
1005                 }
1006                 return c;
1007         }
1008     }

1010     /**
1011      * Helper method for create Javac class
1012      */
1013     protected Class<?> createJavacClass() throws CompilerException {
1014         try {
1015             // look whether JavaC is on Maven's classpath
1016             // return Class.forName( JavacCompiler.JAVAC_CLASSNAME, true, JavacCompiler.class.getClassLoader() );
1017             return JavacCompiler.class.getClassLoader().loadClass(JavacCompiler.JAVAC_CLASSNAME);
1018         } catch (ClassNotFoundException ex) {
1019             // ok
1020         }
1021 
1022         final File toolsJar = new File(System.getProperty("java.home"), "../lib/tools.jar");
1023         if (!toolsJar.exists()) {
1024             throw new CompilerException("tools.jar not found: " + toolsJar);
1025         }
1026 
1027         try {
1028             // Combined classloader with no parent/child relationship, so classes in our classloader
1029             // can reference classes in tools.jar
1030             URL[] originalUrls = ((URLClassLoader) JavacCompiler.class.getClassLoader()).getURLs();
1031             URL[] urls = new URL[originalUrls.length + 1];
1032             urls[0] = toolsJar.toURI().toURL();
1033             System.arraycopy(originalUrls, 0, urls, 1, originalUrls.length);
1034             ClassLoader javacClassLoader = new URLClassLoader(urls);
1035 
1036             final Thread thread = Thread.currentThread();
1037             final ClassLoader contextClassLoader = thread.getContextClassLoader();
1038             thread.setContextClassLoader(javacClassLoader);
1039             try {
1040                 // return Class.forName( JavacCompiler.JAVAC_CLASSNAME, true, javacClassLoader );
1041                 return javacClassLoader.loadClass(JavacCompiler.JAVAC_CLASSNAME);
1042             } finally {
1043                 thread.setContextClassLoader(contextClassLoader);
1044             }
1045         } catch (MalformedURLException ex) {
1046             throw new CompilerException(
1047                     "Could not convert the file reference to tools.jar to a URL, path to tools.jar: '"
1048                             + toolsJar.getAbsolutePath() + "'.",
1049                     ex);
1050         } catch (ClassNotFoundException ex) {
1051             throw new CompilerException(
1052                     "Unable to locate the Javac Compiler in:" + EOL + "  " + toolsJar + EOL
1053                             + "Please ensure you are using JDK 1.4 or above and" + EOL
1054                             + "not a JRE (the com.sun.tools.javac.Main class is required)." + EOL
1055                             + "In most cases you can change the location of your Java" + EOL
1056                             + "installation by setting the JAVA_HOME environment variable.",
1057                     ex);
1058         }
1059     }
