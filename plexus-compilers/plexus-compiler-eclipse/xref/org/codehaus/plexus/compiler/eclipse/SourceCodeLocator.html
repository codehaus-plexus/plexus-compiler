27  import java.io.File;
28  import java.util.HashMap;
29  import java.util.List;
30  import java.util.Map;
31  
32  /**
33   * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
34   */
35  public class SourceCodeLocator
36  {
37      private List<String> sourceRoots;
38  
39      private Map<String, File> cache;
40  
41      public SourceCodeLocator( List<String> sourceRoots )
42      {
43          this.sourceRoots = sourceRoots;
44  
45          cache = new HashMap<>();
46      }
47  
48      public File findSourceCodeForClass( String className )
49      {
50          File f = cache.get( className );
51  
52          if ( f != null )
53          {
54              return f;
55          }
56  
57          String sourceName = className.replace( '.', System.getProperty( "file.separator" ).charAt( 0 ) );
58  
59          sourceName += ".java";
60  
61          f = findInRoots( sourceName );
62  
63          cache.put( className, f );
64  
65          return f;
66      }
67  
68      private File findInRoots( String s )
69      {
70          for ( String root : sourceRoots )
71          {
72              File f = new File( root, s );
73  
74              if ( f.exists() )
75              {
76                  return f;
77              }
78          }
79  
80          return null;
81      }
82  }
