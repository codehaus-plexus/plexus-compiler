105     public CompilerResult performCompile( CompilerConfiguration config )
106         throws CompilerException
107     {
108         File destinationDir = new File( config.getOutputLocation() );
109 
110         if ( !destinationDir.exists() )
111         {
112             destinationDir.mkdirs();
113         }
114 
115         config.setSourceFiles( null );
116 
117         String[] sourceFiles = CSharpCompiler.getSourceFiles( config );
118 
119         if ( sourceFiles.length == 0 )
120         {
121             return new CompilerResult().success( true );
122         }
123 
124         logCompiling( sourceFiles, config );
125 
126         String[] args = buildCompilerArguments( config, sourceFiles );
127 
128         List<CompilerMessage> messages;
129 
130         if ( config.isFork() )
131         {
132             messages =
133                 compileOutOfProcess( config.getWorkingDirectory(), config.getBuildDirectory(), findExecutable( config ),
134                                      args );
135         }
136         else
137         {
138             throw new CompilerException( "This compiler doesn't support in-process compilation." );
139         }
140 
141         return new CompilerResult().compilerMessages( messages );
142     }

154     private Map<String, String> getCompilerArguments( CompilerConfiguration config )
155     {
156         if (compilerArguments != null)
157         {
158             return compilerArguments;
159         }
160         
161         compilerArguments = config.getCustomCompilerArgumentsAsMap();
162         
163         Iterator<String> i = compilerArguments.keySet().iterator();
164         
165         while ( i.hasNext() ) 
166         {
167             String orig = i.next();
168             String v = compilerArguments.get( orig );
169             if ( orig.contains( ":" ) && v == null ) 
170             {
171                 String[] arr = orig.split( ":" );
172                 i.remove();
173                 String k = arr[0];
174                 v = arr[1];
175                 compilerArguments.put( k, v );
176                 if ( config.isDebug() )
177                 {
178                     System.out.println( "transforming argument from " + orig + " to " + k + " = [" + v + "]" );
179                 }
180             }
181         }
182         
183         config.setCustomCompilerArgumentsAsMap( compilerArguments );
184         
185         return compilerArguments;
186     }

188     private String findExecutable( CompilerConfiguration config )
189     {
190         String executable = config.getExecutable();
191 
192         if ( !StringUtils.isEmpty( executable ) )
193         {
194             return executable;
195         }
196 
197         if ( Os.isFamily( "windows" ) )
198         {
199             return "csc";
200         }
201 
202         return "mcs";
203     }

248     private String[] buildCompilerArguments( CompilerConfiguration config, String[] sourceFiles )
249         throws CompilerException
250     {
251         List<String> args = new ArrayList<>();
252 
253         if ( config.isDebug() )
254         {
255             args.add( "/debug+" );
256         }
257         else
258         {
259             args.add( "/debug-" );
260         }
261 
262         // config.isShowWarnings()
263         // config.getSourceVersion()
264         // config.getTargetVersion()
265         // config.getSourceEncoding()
266 
267         // ----------------------------------------------------------------------
268         //
269         // ----------------------------------------------------------------------
270 
271         for ( String element : config.getClasspathEntries() )
272         {
273             File f = new File( element );
274 
275             if ( !f.isFile() )
276             {
277                 continue;
278             }
279             
280             if (element.endsWith(JAR_SUFFIX)) {
281                 try 
282                 {
283                     File dllDir = new File(element + NET_SUFFIX);
284                     if (!dllDir.exists())
285                     {
286                         dllDir.mkdir();
287                     }
288                     JarUtil.extract(dllDir, new File(element));
289                     for (String tmpfile : dllDir.list()) 
290                     {
291                         if ( tmpfile.endsWith(DLL_SUFFIX) )
292                         {
293                             String dll = Paths.get(dllDir.getAbsolutePath(), tmpfile).toString();
294                             args.add( "/reference:\"" + dll + "\"" );
295                         }
296                     }
297                 }
298                 catch ( IOException e )
299                 {
300                     throw new CompilerException( e.toString(), e );
301                 }
302             }
303             else
304             {
305                 args.add( "/reference:\"" + element + "\"" );
306             }
307         }
308 
309         // ----------------------------------------------------------------------
310         // Main class
311         // ----------------------------------------------------------------------
312 
313         Map<String, String> compilerArguments = getCompilerArguments( config );
314 
315         String mainClass = compilerArguments.get( "-main" );
316 
317         if ( !StringUtils.isEmpty( mainClass ) )
318         {
319             args.add( "/main:" + mainClass );
320         }
321 
322         // ----------------------------------------------------------------------
323         // Xml Doc output
324         // ----------------------------------------------------------------------
325 
326         String doc = compilerArguments.get( "-doc" );
327 
328         if ( !StringUtils.isEmpty( doc ) )
329         {
330             args.add( "/doc:" + new File( config.getOutputLocation(),
331                                           config.getOutputFileName() + ".xml" ).getAbsolutePath() );
332         }
333 
334         // ----------------------------------------------------------------------
335         // Xml Doc output
336         // ----------------------------------------------------------------------
337 
338         String nowarn = compilerArguments.get( "-nowarn" );
339 
340         if ( !StringUtils.isEmpty( nowarn ) )
341         {
342             args.add( "/nowarn:" + nowarn );
343         }
344 
345         // ----------------------------------------------------------------------
346         // Out - Override output name, this is required for generating the unit test dll
347         // ----------------------------------------------------------------------
348 
349         String out = compilerArguments.get( "-out" );
350 
351         if ( !StringUtils.isEmpty( out ) )
352         {
353             args.add( "/out:" + new File( config.getOutputLocation(), out ).getAbsolutePath() );
354         }
355         else
356         {
357             args.add( "/out:" + new File( config.getOutputLocation(), getOutputFile( config ) ).getAbsolutePath() );
358         }
359 
360         // ----------------------------------------------------------------------
361         // Resource File - compile in a resource file into the assembly being created
362         // ----------------------------------------------------------------------
363         String resourcefile = compilerArguments.get( "-resourcefile" );
364 
365         if ( !StringUtils.isEmpty( resourcefile ) )
366         {
367             String resourceTarget = compilerArguments.get( "-resourcetarget" );
368             args.add( "/res:" + new File( resourcefile ).getAbsolutePath() + "," + resourceTarget );
369         }
370 
371         // ----------------------------------------------------------------------
372         // Target - type of assembly to produce, lib,exe,winexe etc... 
373         // ----------------------------------------------------------------------
374 
375         String target = compilerArguments.get( "-target" );
376 
377         if ( StringUtils.isEmpty( target ) )
378         {
379             args.add( "/target:library" );
380         }
381         else
382         {
383             args.add( "/target:" + target );
384         }
385 
386         // ----------------------------------------------------------------------
387         // remove MS logo from output (not applicable for mono)
388         // ----------------------------------------------------------------------
389         String nologo = compilerArguments.get( "-nologo" );
390 
391         if ( !StringUtils.isEmpty( nologo ) )
392         {
393             args.add( "/nologo" );
394         }
395 
396         // ----------------------------------------------------------------------
397         // add any resource files
398         // ----------------------------------------------------------------------
399         this.addResourceArgs( config, args );
400 
401         // ----------------------------------------------------------------------
402         // add source files
403         // ----------------------------------------------------------------------
404         for ( String sourceFile : sourceFiles )
405         {
406             args.add( sourceFile );
407         }
408 
409         return args.toArray( new String[args.size()] );
410     }

412     private void addResourceArgs( CompilerConfiguration config, List<String> args )
413     {
414         File filteredResourceDir = this.findResourceDir( config );
415         if ( ( filteredResourceDir != null ) && filteredResourceDir.exists() )
416         {
417             DirectoryScanner scanner = new DirectoryScanner();
418             scanner.setBasedir( filteredResourceDir );
419             scanner.setIncludes( DEFAULT_INCLUDES );
420             scanner.addDefaultExcludes();
421             scanner.scan();
422 
423             List<String> includedFiles = Arrays.asList( scanner.getIncludedFiles() );
424             for ( String name : includedFiles )
425             {
426                 File filteredResource = new File( filteredResourceDir, name );
427                 String assemblyResourceName = this.convertNameToAssemblyResourceName( name );
428                 String argLine = "/resource:\"" + filteredResource + "\",\"" + assemblyResourceName + "\"";
429                 if ( config.isDebug() )
430                 {
431                     System.out.println( "adding resource arg line:" + argLine );
432                 }
433                 args.add( argLine );
434 
435             }
436         }
437     }

439     private File findResourceDir( CompilerConfiguration config )
440     {
441         if ( config.isDebug() )
442         {
443             System.out.println( "Looking for resourcesDir" );
444         }
445         
446         Map<String, String> compilerArguments = getCompilerArguments( config );
447         
448         String tempResourcesDirAsString = compilerArguments.get( "-resourceDir" );
449         File filteredResourceDir = null;
450         if ( tempResourcesDirAsString != null )
451         {
452             filteredResourceDir = new File( tempResourcesDirAsString );
453             if ( config.isDebug() )
454             {
455                 System.out.println( "Found resourceDir at: " + filteredResourceDir.toString() );
456             }
457         }
458         else
459         {
460             if ( config.isDebug() )
461             {
462                 System.out.println( "No resourceDir was available." );
463             }
464         }
465         return filteredResourceDir;
466     }

473     @SuppressWarnings( "deprecation" )
474     private List<CompilerMessage> compileOutOfProcess( File workingDirectory, File target, String executable,
475                                                        String[] args )
476         throws CompilerException
477     {
478         // ----------------------------------------------------------------------
479         // Build the @arguments file
480         // ----------------------------------------------------------------------
481 
482         File file;
483 
484         PrintWriter output = null;
485 
486         try
487         {
488             file = new File( target, ARGUMENTS_FILE_NAME );
489 
490             output = new PrintWriter( new FileWriter( file ) );
491 
492             for ( String arg : args )
493             {
494                 output.println( arg );
495             }
496         }
497         catch ( IOException e )
498         {
499             throw new CompilerException( "Error writing arguments file.", e );
500         }
501         finally
502         {
503             IOUtil.close( output );
504         }
505 
506         // ----------------------------------------------------------------------
507         // Execute!
508         // ----------------------------------------------------------------------
509 
510         Commandline cli = new Commandline();
511 
512         cli.setWorkingDirectory( workingDirectory.getAbsolutePath() );
513 
514         cli.setExecutable( executable );
515 
516         cli.createArgument().setValue( "@" + file.getAbsolutePath() );
517 
518         Writer stringWriter = new StringWriter();
519 
520         StreamConsumer out = new WriterStreamConsumer( stringWriter );
521 
522         StreamConsumer err = new WriterStreamConsumer( stringWriter );
523 
524         int returnCode;
525 
526         List<CompilerMessage> messages;
527 
528         try
529         {
530             returnCode = CommandLineUtils.executeCommandLine( cli, out, err );
531 
532             messages = parseCompilerOutput( new BufferedReader( new StringReader( stringWriter.toString() ) ) );
533         }
534         catch ( CommandLineException | IOException e )
535         {
536             throw new CompilerException( "Error while executing the external compiler.", e );
537         }
538 
539         if ( returnCode != 0 && messages.isEmpty() )
540         {
541             // TODO: exception?
542             messages.add( new CompilerMessage(
543                 "Failure executing the compiler, but could not parse the error:" + EOL + stringWriter.toString(),
544                 true ) );
545         }
546 
547         return messages;
548     }

550     public static List<CompilerMessage> parseCompilerOutput( BufferedReader bufferedReader )
551         throws IOException
552     {
553         List<CompilerMessage> messages = new ArrayList<>();
554 
555         String line = bufferedReader.readLine();
556 
557         while ( line != null )
558         {
559             CompilerMessage compilerError = DefaultCSharpCompilerParser.parseLine( line );
560 
561             if ( compilerError != null )
562             {
563                 messages.add( compilerError );
564             }
565 
566             line = bufferedReader.readLine();
567         }
568 
569         return messages;
570     }

584     private String getTypeExtension( CompilerConfiguration configuration )
585         throws CompilerException
586     {
587         String type = getType( configuration.getCustomCompilerArgumentsAsMap() );
588 
589         if ( "exe".equals( type ) || "winexe".equals( type ) )
590         {
591             return "exe";
592         }
593 
594         if ( "library".equals( type ) || "module".equals( type ) )
595         {
596             return "dll";
597         }
598 
599         throw new CompilerException( "Unrecognized type '" + type + "'." );
600     }

602     // added for debug purposes.... 
603     protected static String[] getSourceFiles( CompilerConfiguration config )
604     {
605         Set<String> sources = new HashSet<>();
606 
607         //Set sourceFiles = null;
608         //was:
609         Set<File> sourceFiles = config.getSourceFiles();
610 
611         if ( sourceFiles != null && !sourceFiles.isEmpty() )
612         {
613             for ( File sourceFile : sourceFiles )
614             {
615                 sources.add( sourceFile.getAbsolutePath() );
616             }
617         }
618         else
619         {
620             for ( String sourceLocation : config.getSourceLocations() )
621             {
622                 if (!new File(sourceLocation).exists())
623                 {
624                     if ( config.isDebug() )
625                     {
626                         System.out.println( "Ignoring not found sourceLocation at: " + sourceLocation );
627                     }
628                     continue;
629                 }
630                 sources.addAll( getSourceFilesForSourceRoot( config, sourceLocation ) );
631             }
632         }
633 
634         String[] result;
635 
636         if ( sources.isEmpty() )
637         {
638             result = new String[0];
639         }
640         else
641         {
642             result = sources.toArray( new String[sources.size()] );
643         }
644 
645         return result;
646     }

648     protected static Set<String> getSourceFilesForSourceRoot( CompilerConfiguration config, String sourceLocation )
649     {
650         DirectoryScanner scanner = new DirectoryScanner();
651 
652         scanner.setBasedir( sourceLocation );
653 
654         Set<String> includes = config.getIncludes();
655 
656         if ( includes != null && !includes.isEmpty() )
657         {
658             String[] inclStrs = includes.toArray( new String[includes.size()] );
659             scanner.setIncludes( inclStrs );
660         }
661         else
662         {
663             scanner.setIncludes( new String[]{ "**/*.cs" } );
664         }
665 
666         Set<String> excludes = config.getExcludes();
667 
668         if ( excludes != null && !excludes.isEmpty() )
669         {
670             String[] exclStrs = excludes.toArray( new String[excludes.size()] );
671             scanner.setIncludes( exclStrs );
672         }
673 
674         scanner.scan();
675 
676         String[] sourceDirectorySources = scanner.getIncludedFiles();
677 
678         Set<String> sources = new HashSet<>();
679 
680         for ( String source : sourceDirectorySources )
681         {
682             File f = new File( sourceLocation, source );
683 
684             sources.add( f.getPath() );
685         }
686 
687         return sources;
688     }
